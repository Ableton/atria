130  template <typename TestFnT>
131  void run(std::string name, TestFnT test_fn)
132  {
133  if (settings_.verbose)
134  std::cout << " ... " << name << std::endl;
135 
136  auto minimum = [] (double x, double y) { return std::min(x, y); };
137  auto iteration = [&] (unsigned) { unoptimize(test_fn()); };
138  auto measure = [&] (unsigned) {
139  return timeit([&] {
140  boost::for_each(boost::irange(0u, settings_.iterations), iteration);
141  }).count();
142  };
143 
144  results_.insert(
145  std::make_tuple(
146  xform::transduce(
147  xform::map(measure),
148  minimum,
149  std::numeric_limits<double>::max(),
150  boost::irange(0u, settings_.measurements)),
151  std::move(name)));
152  }
153 
154 private:
155  std::string name_;
156  benchmark_settings settings_;
157  std::set<std::tuple<double, std::string> > results_;
158 };
